**Goal:** A complete, end-to-end run lifecycle. A user creates a `BenchmarkRun` CR; the operator allocates pods (gVisor submission + reference-oracle); the bot-coordinator emits a deterministic schedule; bot-workers fire orders at the fairness-gateway, which stamps `(platform_seq, platform_ts)`, strips identity, and tees to both submission and oracle; both reply with acks/fills; the telemetry-sidecar captures every event to a local file (Redpanda comes in Phase 3).

```rust
//! Primitive types for the order book and matcher.
//!
//! All prices are integer ticks; all quantities are unsigned. No floats.

use serde::{Deserialize, Serialize};

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, PartialOrd, Ord, Serialize, Deserialize)]
pub struct Price(pub i64);

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, PartialOrd, Ord, Serialize, Deserialize)]
pub struct Qty(pub u64);

impl Qty {
    #[must_use] pub const fn zero() -> Self { Self(0) }
    pub fn checked_sub(self, rhs: Qty) -> Option<Qty> { self.0.checked_sub(rhs.0).map(Qty) }
    pub fn min(self, rhs: Qty) -> Qty { Qty(self.0.min(rhs.0)) }
}

#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize)]
pub enum Side { Bid, Ask }

#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize)]
pub enum TimeInForce { Gtc, Ioc, Fok }

#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize)]
pub enum OrderType {
    Limit { price: Price, tif: TimeInForce },
    Market,
}

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, Serialize, Deserialize)]
pub struct ClientOrderId(pub u128);

impl ClientOrderId {
    #[must_use] pub const fn new(bot_id: u64, local_seq: u64) -> Self {
        Self((bot_id as u128) << 64 | (local_seq as u128))
    }
    #[must_use] pub fn bot_id(self) -> u64 { (self.0 >> 64) as u64 }
    #[must_use] pub fn local_seq(self) -> u64 { (self.0 & 0xFFFF_FFFF_FFFF_FFFF) as u64 }
}

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, Serialize, Deserialize)]
pub struct SessionToken(pub [u8; 32]);

#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct Order {
    pub platform_seq:   u64,
    pub platform_ts_ns: u64,
    pub client_order_id: ClientOrderId,
    pub session_token:  SessionToken,
```

```rust
//! Order book — BTreeMap<Price, VecDeque<Resting>> per side.

use std::collections::{BTreeMap, HashMap, VecDeque};

use crate::types::*;

#[derive(Clone, Debug)]
pub struct Resting {
    pub platform_seq:   u64,
    pub platform_ts_ns: u64,
    pub client_order_id: ClientOrderId,
    pub session_token:  SessionToken,
    pub qty_remaining:  Qty,
}

#[derive(Default, Debug)]
pub struct OrderBook {
    bids: BTreeMap<Price, VecDeque<Resting>>,
    asks: BTreeMap<Price, VecDeque<Resting>>,
    by_id: HashMap<ClientOrderId, (Side, Price)>,
    next_trade_id: u64,
}

impl OrderBook {
    #[must_use] pub fn new() -> Self { Self::default() }

    /// Highest bid price with at least one resting order, or None.
    #[must_use]
    pub fn best_bid(&self) -> Option<Price> {
        self.bids.iter().next_back().map(|(p, _)| *p)
    }

    /// Lowest ask price with at least one resting order, or None.
    #[must_use]
    pub fn best_ask(&self) -> Option<Price> {
        self.asks.iter().next().map(|(p, _)| *p)
    }

    /// Insert a resting order. The order is appended to the price level's queue.
    pub fn rest(&mut self, o: Order) {
        let (side, price) = match (o.side, o.kind) {
            (s, OrderType::Limit { price, .. }) => (s, price),
            _ => return, // market orders never rest
        };
        let book = match side { Side::Bid => &mut self.bids, Side::Ask => &mut self.asks };
        let q = book.entry(price).or_default();
        q.push_back(Resting {
            platform_seq:   o.platform_seq,
            platform_ts_ns: o.platform_ts_ns,
            client_order_id: o.client_order_id,
            session_token:  o.session_token,
            qty_remaining:  o.qty,
        });
        self.by_id.insert(o.client_order_id, (side, price));
    }

    /// Cancel a resting order. Returns true if cancelled.
    pub fn cancel(&mut self, id: ClientOrderId) -> bool {
        let Some((side, price)) = self.by_id.remove(&id) else { return false };
        let book = match side { Side::Bid => &mut self.bids, Side::Ask => &mut self.asks };
        let Some(q) = book.get_mut(&price) else { return false };
        let mut idx = None;
        for (i, r) in q.iter().enumerate() {
            if r.client_order_id == id { idx = Some(i); break; }
        }
        if let Some(i) = idx {
            q.remove(i);
            if q.is_empty() { book.remove(&price); }
            true
        } else { false }
    }

    /// Returns the next trade id and advances the counter.
    pub(crate) fn next_trade_id(&mut self) -> u64 {
        let id = self.next_trade_id;
        self.next_trade_id += 1;
        id
    }

    /// Price level access for the matcher.
    pub(crate) fn level_mut(&mut self, side: Side, price: Price) -> Option<&mut VecDeque<Resting>> {
        match side { Side::Bid => self.bids.get_mut(&price), Side::Ask => self.asks.get_mut(&price) }
    }

    /// Yield best price + queue for matching against `taker_side`.
    pub(crate) fn best_for(&mut self, taker_side: Side) -> Option<(Price, &mut VecDeque<Resting>)> {
        match taker_side {
            // Bid taker matches the lowest ask first.
            Side::Bid => self.asks.iter_mut().next().map(|(p, q)| (*p, q)),
            // Ask taker matches the highest bid first.
            Side::Ask => self.bids.iter_mut().next_back().map(|(p, q)| (*p, q)),
        }
    }

    pub(crate) fn drop_level_if_empty(&mut self, side: Side, price: Price) {
        let book = match side { Side::Bid => &mut self.bids, Side::Ask => &mut self.asks };
        if book.get(&price).map_or(false, |q| q.is_empty()) {
            book.remove(&price);
        }
    }

    pub(crate) fn forget_id(&mut self, id: ClientOrderId) { let _ = self.by_id.remove(&id); }
}
```

```rust
//! Price-time priority matcher.

use crate::book::OrderBook;
use crate::types::*;

impl OrderBook {
    /// Match a taker order against the book. Returns fills produced.
    /// Mutates the book according to GTC/IOC/FOK semantics.
    pub fn match_order(&mut self, taker: Order) -> Vec<Fill> {
        let mut fills = Vec::new();
        let mut remaining = taker.qty;

        // FOK pre-flight: can we fill in full at acceptable prices?
        if let OrderType::Limit { price: lim, tif: TimeInForce::Fok } = taker.kind {
            if !self.can_fill_full(taker.side, remaining, lim) { return fills; }
        }

        // Walk levels until exhausted, no longer crossing, or hit FOK rejection.
        while remaining.0 > 0 {
            let Some((price, queue)) = self.best_for(taker.side) else { break };
            if !crosses(taker.kind, taker.side, price) { break; }

            while remaining.0 > 0 {
                let Some(front) = queue.front_mut() else { break };
                let traded = remaining.min(front.qty_remaining);
                fills.push(Fill {
                    trade_id:           0, // patched below — borrow scope
                    platform_seq_taker: taker.platform_seq,
                    platform_seq_maker: front.platform_seq,
                    price,
                    qty:                traded,
                    ts_ns:              taker.platform_ts_ns,
                });
                remaining = remaining.checked_sub(traded).expect("checked above");
                front.qty_remaining = front.qty_remaining.checked_sub(traded).expect("checked");
                if front.qty_remaining.0 == 0 {
                    let id = front.client_order_id;
                    queue.pop_front();
                    self.forget_id(id);
                }
            }
            self.drop_level_if_empty(taker.side, price);
        }
        // Patch trade_ids monotonically.
        for f in &mut fills { f.trade_id = self.next_trade_id(); }

        // Rest remainder according to TIF.
        if remaining.0 > 0 {
            match taker.kind {
                OrderType::Limit { tif: TimeInForce::Gtc, .. } => {
                    let mut rest = taker.clone();
                    rest.qty = remaining;
                    self.rest(rest);
                }
                OrderType::Limit { tif: TimeInForce::Ioc | TimeInForce::Fok, .. } => {} // discard
                OrderType::Market => {} // discard
            }
        }

        fills
    }

    /// True if a sweep starting at `lim` could fill `qty` units. Used for FOK pre-flight.
    fn can_fill_full(&self, taker_side: Side, qty: Qty, lim: Price) -> bool {
        let mut remaining = qty.0;
        let levels: Box<dyn Iterator<Item = (&Price, &std::collections::VecDeque<crate::book::Resting>)>> =
            match taker_side {
                Side::Bid => Box::new(self.asks.iter()),
                Side::Ask => Box::new(self.bids.iter().rev()),
            };
        for (p, q) in levels {
            if !crosses_price(taker_side, *p, lim) { break; }
            for r in q { remaining = remaining.saturating_sub(r.qty_remaining.0); if remaining == 0 { return true; } }
        }
        false
    }
}

fn crosses(kind: OrderType, taker_side: Side, level: Price) -> bool {
    match kind {
        OrderType::Limit { price, .. } => crosses_price(taker_side, level, price),
        OrderType::Market => true,
    }
}

fn crosses_price(taker_side: Side, level: Price, taker_price: Price) -> bool {
    match taker_side { Side::Bid => level <= taker_price, Side::Ask => level >= taker_price }
}
```

```rust
use matching_engine::*;
use proptest::prelude::*;

fn arb_order(seq: u64) -> impl Strategy<Value = Order> {
    (
        any::<bool>(),                                                // side
        1u64..200,                                                    // qty
        50i64..150,                                                   // price
        prop_oneof![Just(TimeInForce::Gtc), Just(TimeInForce::Ioc), Just(TimeInForce::Fok)],
    ).prop_map(move |(b, q, p, tif)| Order {
        platform_seq:    seq,
        platform_ts_ns:  seq,
        client_order_id: ClientOrderId::new(1, seq),
        session_token:   SessionToken([0; 32]),
        side:            if b { Side::Bid } else { Side::Ask },
        qty:             Qty(q),
        kind:            OrderType::Limit { price: Price(p), tif },
    })
}

fn arb_sequence(n: usize) -> impl Strategy<Value = Vec<Order>> {
    prop::collection::vec(any::<bool>(), n)
        .prop_flat_map(|_| {
            // generate exactly n orders, seq = 1..=n, deterministic
            let strategies: Vec<_> = (1..=(n as u64)).map(arb_order).collect();
            strategies
        })
}

proptest! {
    #![proptest_config(ProptestConfig { cases: 256, .. ProptestConfig::default() })]

    #[test]
    fn fill_conservation(orders in arb_sequence(30)) {
        let mut b = OrderBook::new();
        let mut buy_filled  = 0u64;
        let mut sell_filled = 0u64;
        for o in &orders {
            for f in b.match_order(o.clone()) {
                match o.side {
                    Side::Bid => buy_filled  += f.qty.0,
                    Side::Ask => sell_filled += f.qty.0,
                }
                prop_assert!(f.qty.0 > 0);
                prop_assert!(f.price.0 > 0);
            }
        }
        prop_assert_eq!(buy_filled, sell_filled);
    }

    #[test]
    fn no_phantom_makers(orders in arb_sequence(30)) {
        let mut b = OrderBook::new();
        let mut seen_seqs: std::collections::HashSet<u64> = std::collections::HashSet::new();
        for o in &orders {
            seen_seqs.insert(o.platform_seq);
            for f in b.match_order(o.clone()) {
                prop_assert!(seen_seqs.contains(&f.platform_seq_maker),
                  "phantom maker seq {} not yet seen at taker seq {}", f.platform_seq_maker, f.platform_seq_taker);
                prop_assert!(seen_seqs.contains(&f.platform_seq_taker));
            }
        }
    }

    #[test]
    fn trade_ids_monotonic(orders in arb_sequence(30)) {
        let mut b = OrderBook::new();
        let mut last = 0u64;
        for o in &orders {
            for f in b.match_order(o.clone()) {
                prop_assert!(f.trade_id >= last);
                last = f.trade_id;
            }
        }
    }
}
```

```rust
//! Deterministic serialization of the OrderBook for crash recovery.

use crate::book::OrderBook;
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize)]
struct Snap {
    bids: Vec<(i64, Vec<RestingSnap>)>,
    asks: Vec<(i64, Vec<RestingSnap>)>,
    next_trade_id: u64,
}
#[derive(Serialize, Deserialize)]
struct RestingSnap {
    platform_seq: u64,
    platform_ts_ns: u64,
    client_order_id_hi: u64,
    client_order_id_lo: u64,
    session_token: [u8; 32],
    qty_remaining: u64,
}

impl OrderBook {
    pub fn snapshot(&self) -> Vec<u8> {
        let snap = Snap {
            bids: self.bids.iter().map(|(p, q)| (p.0, q.iter().map(snap_one).collect())).collect(),
            asks: self.asks.iter().map(|(p, q)| (p.0, q.iter().map(snap_one).collect())).collect(),
            next_trade_id: self.next_trade_id,
        };
        bincode::serde::encode_to_vec(&snap, bincode::config::standard()).expect("infallible serialize")
    }
    pub fn restore(bytes: &[u8]) -> Result<Self, bincode::error::DecodeError> {
        let (snap, _): (Snap, _) = bincode::serde::decode_from_slice(bytes, bincode::config::standard())?;
        let mut b = OrderBook::new();
        for (p, q) in snap.bids {
            for r in q { b.bids.entry(crate::types::Price(p)).or_default().push_back(restore_one(r, crate::types::Side::Bid)); }
        }
        for (p, q) in snap.asks {
            for r in q { b.asks.entry(crate::types::Price(p)).or_default().push_back(restore_one(r, crate::types::Side::Ask)); }
        }
        b.next_trade_id = snap.next_trade_id;
        // rebuild by_id
        for (p, q) in &b.bids { for r in q { b.by_id.insert(r.client_order_id, (crate::types::Side::Bid, *p)); } }
        for (p, q) in &b.asks { for r in q { b.by_id.insert(r.client_order_id, (crate::types::Side::Ask, *p)); } }
        Ok(b)
    }
}
fn snap_one(r: &crate::book::Resting) -> RestingSnap {
    let id = r.client_order_id.0;
    RestingSnap {
        platform_seq: r.platform_seq, platform_ts_ns: r.platform_ts_ns,
        client_order_id_hi: (id >> 64) as u64, client_order_id_lo: id as u64,
        session_token: r.session_token.0, qty_remaining: r.qty_remaining.0,
    }
}
fn restore_one(s: RestingSnap, _side: crate::types::Side) -> crate::book::Resting {
    crate::book::Resting {
        platform_seq: s.platform_seq, platform_ts_ns: s.platform_ts_ns,
        client_order_id: crate::types::ClientOrderId(((s.client_order_id_hi as u128) << 64) | (s.client_order_id_lo as u128)),
        session_token: crate::types::SessionToken(s.session_token),
        qty_remaining: crate::types::Qty(s.qty_remaining),
    }
}
```

```rust
//! Monotonic stamp generator. Persists high-water-mark to disk on graceful shutdown.

use std::sync::atomic::{AtomicU64, Ordering};
use std::time::{SystemTime, UNIX_EPOCH};

pub struct Clock {
    next_seq: AtomicU64,
    boot_ts_ns: u64,
    start_instant: std::time::Instant,
}

impl Clock {
    pub fn new(starting_seq: u64) -> Self {
        let boot_ts_ns = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_nanos() as u64;
        Self { next_seq: AtomicU64::new(starting_seq), boot_ts_ns, start_instant: std::time::Instant::now() }
    }

    /// Reserves `n` stamps, returns the first seq + ts.
    pub fn reserve(&self, n: u64) -> (u64, u64) {
        let seq = self.next_seq.fetch_add(n, Ordering::SeqCst);
        let elapsed = self.start_instant.elapsed().as_nanos() as u64;
        (seq, self.boot_ts_ns + elapsed)
    }

    pub fn high_watermark(&self) -> u64 { self.next_seq.load(Ordering::SeqCst) }
}
```

```rust
use ironbook_proto::gen::ironbook::v1::*;
use matching_engine::*;
use parking_lot::Mutex;
use tonic::{Request, Response, Status};

pub struct Oracle { pub book: Mutex<OrderBook> }

#[tonic::async_trait]
impl order_intake_server::OrderIntake for Oracle {
    async fn submit(&self, req: Request<NormalizedOrder>) -> Result<Response<Reply>, Status> {
        let n = req.into_inner();
        let order = match wire_to_order(&n) {
            Ok(o) => o,
            Err(e) => return Ok(Response::new(Reply { ack: Some(Ack { platform_seq: n.platform_seq, status: 1, message: e, ack_ts_ns: now_ns() }), fills: vec![] })),
        };
        let fills = self.book.lock().match_order(order.clone());
        let wire_fills = fills.into_iter().map(|f| Fill {
            trade_id: f.trade_id,
            platform_seq_taker: f.platform_seq_taker,
            platform_seq_maker: f.platform_seq_maker,
            price: f.price.0, qty: f.qty.0, ts_ns: f.ts_ns,
        }).collect();
        Ok(Response::new(Reply {
            ack: Some(Ack { platform_seq: order.platform_seq, status: 0, message: String::new(), ack_ts_ns: now_ns() }),
            fills: wire_fills,
        }))
    }
}

fn wire_to_order(n: &NormalizedOrder) -> Result<Order, String> {
    let side = match Side::try_from(n.side).map_err(|_| "bad side".to_string())? {
        Side::Bid => matching_engine::Side::Bid,
        Side::Ask => matching_engine::Side::Ask,
        _ => return Err("unspecified side".into()),
    };
    let kind = match OrderType::try_from(n.order_type).map_err(|_| "bad order type".to_string())? {
        OrderType::Limit  => matching_engine::OrderType::Limit { price: matching_engine::Price(n.price), tif: tif_from(n.tif)? },
        OrderType::Market => matching_engine::OrderType::Market,
        _ => return Err("unspecified order type".into()),
    };
    Ok(matching_engine::Order {
        platform_seq: n.platform_seq, platform_ts_ns: n.platform_ts_ns,
        client_order_id: matching_engine::ClientOrderId(u128::from_be_bytes(n.client_order_id.clone().try_into().map_err(|_| "bad coid")?)),
        session_token: matching_engine::SessionToken(n.session_token.clone().try_into().map_err(|_| "bad token")?),
        side, qty: matching_engine::Qty(n.qty), kind,
    })
}

fn tif_from(v: i32) -> Result<matching_engine::TimeInForce, String> {
    match TimeInForce::try_from(v).map_err(|_| "bad tif".to_string())? {
        TimeInForce::Gtc => Ok(matching_engine::TimeInForce::Gtc),
        TimeInForce::Ioc => Ok(matching_engine::TimeInForce::Ioc),
        TimeInForce::Fok => Ok(matching_engine::TimeInForce::Fok),
        _ => Err("unspecified tif".into()),
    }
}
fn now_ns() -> u64 { std::time::SystemTime::now().duration_since(std::time::UNIX_EPOCH).unwrap().as_nanos() as u64 }
```

```go
type Stamper struct {
	cli       pb.TimeServiceClient
	mu        sync.Mutex
	nextSeq   uint64
	endSeq    uint64
	originTs  uint64
	startWall time.Time
}

func NewStamper(conn *grpc.ClientConn) *Stamper { return &Stamper{cli: pb.NewTimeServiceClient(conn)} }

func (s *Stamper) Next(ctx context.Context) (seq uint64, tsNs uint64, err error) {
	s.mu.Lock(); defer s.mu.Unlock()
	if s.nextSeq >= s.endSeq {
		resp, e := s.cli.NextStamps(ctx, &pb.NextStampsRequest{BatchSize: 10000})
		if e != nil { return 0, 0, e }
		s.nextSeq, s.endSeq, s.originTs = resp.FirstSeq, resp.FirstSeq+uint64(resp.BatchSize), resp.FirstTsNs
		s.startWall = time.Now()
	}
	seq = s.nextSeq; s.nextSeq++
	tsNs = s.originTs + uint64(time.Since(s.startWall).Nanoseconds())
	return
}
```

```go
package gateway

import (
	"context"
	"errors"
	"sync"

	pb "github.com/<owner>/IronBook/pkg/proto/ironbook/v1"
)

type Fork struct {
	Submission pb.OrderIntakeClient
	Oracle     pb.OrderIntakeClient
	Sink       Sink   // local-file sink today; Redpanda in Phase 3
}

type Sink interface { OnIn(o *pb.NormalizedOrder); OnReply(o *pb.NormalizedOrder, source string, r *pb.Reply) }

func (f *Fork) Process(ctx context.Context, o *pb.NormalizedOrder) (*pb.Reply, error) {
	f.Sink.OnIn(o)
	var subR, orR *pb.Reply
	var subE, orE error
	wg := sync.WaitGroup{}; wg.Add(2)
	go func() { defer wg.Done(); subR, subE = f.Submission.Submit(ctx, o); if subE == nil { f.Sink.OnReply(o, "submission", subR) } }()
	go func() { defer wg.Done(); orR,  orE  = f.Oracle.Submit(ctx, o);     if orE  == nil { f.Sink.OnReply(o, "oracle",     orR)  } }()
	wg.Wait()
	if subE != nil || orE != nil {
		return nil, errors.Join(subE, orE)
	}
	// Bot sees submission's reply (oracle is parallel ground truth).
	return subR, nil
}
```

```go
	Price         int64  `json:"price"`
	OrderType     string `json:"order_type"` // "limit"|"market"
	TIF           string `json:"tif"`        // "gtc"|"ioc"|"fok"
}

func (s *RESTServer) HandleOrder(w http.ResponseWriter, r *http.Request) {
	var ro RESTOrder
	if err := json.NewDecoder(r.Body).Decode(&ro); err != nil { http.Error(w,err.Error(),400); return }
	ctx := r.Context()
	seq, ts, err := s.Stamper.Next(ctx)
	if err != nil { http.Error(w,err.Error(),502); return }
	o := &pb.NormalizedOrder{
		PlatformSeq: seq, PlatformTsNs: ts,
		ClientOrderId: packCOID(ro.BotID, ro.LocalSeq),
		SessionToken:  s.tokenFor(ro.BotID),
		Side:          sideFromString(ro.Side),
		Qty:           ro.Qty,
		Price:         ro.Price,
		OrderType:     orderTypeFromString(ro.OrderType),
		Tif:           tifFromString(ro.TIF),
	}
	rep, err := s.Fork.Process(ctx, o)
	if err != nil { http.Error(w,err.Error(),502); return }
	w.Header().Set("Content-Type", "application/json")
	_ = json.NewEncoder(w).Encode(rep)
}

func (s *RESTServer) tokenFor(botID uint64) []byte {
	h := sha256.New(); h.Write(s.RunSecret); h.Write([]byte(hex.EncodeToString(uint64bytes(botID))))
	return h.Sum(nil)
}
func packCOID(b, l uint64) []byte { /* big-endian 16 bytes */ out := make([]byte,16); for i:=0;i<8;i++{out[7-i]=byte(b>>(i*8))}; for i:=0;i<8;i++{out[15-i]=byte(l>>(i*8))}; return out }
func uint64bytes(x uint64) []byte { out := make([]byte,8); for i:=0;i<8;i++{out[7-i]=byte(x>>(i*8))}; return out }
func sideFromString(s string) int32      { if s=="bid" { return int32(pb.Side_BID) }; return int32(pb.Side_ASK) }
func orderTypeFromString(s string) int32 { if s=="market" { return int32(pb.OrderType_MARKET) }; return int32(pb.OrderType_LIMIT) }
func tifFromString(s string) int32       { switch s { case "ioc": return int32(pb.TimeInForce_IOC); case "fok": return int32(pb.TimeInForce_FOK); default: return int32(pb.TimeInForce_GTC) } }
```

```go
	if err := r.Get(ctx, req.NamespacedName, &br); err != nil {

	}

	switch br.Status.Phase {
	case "":            return r.toPending(ctx, &br)
	case "PENDING":     return r.toAllocating(ctx, &br)
	case "ALLOCATING":  return r.toPriming(ctx, &br)
	case "PRIMING":     return r.toRunning(ctx, &br)
	case "RUNNING":     return r.observeRunning(ctx, &br)
	case "DRAINING":    return r.toComplete(ctx, &br)

	     "INVALID",
```

`coordinator/schedule.go`:
```go
package coordinator

import (
	"crypto/sha256"
	"encoding/binary"
	"hash/fnv"
	"math/rand"
)

type Event struct {
	OffsetNs uint64
	BotID    uint64
	Order    OrderSpec
}
type OrderSpec struct {
	Side, OrderType, TIF string
	Qty   uint64
	Price int64
}

func Compile(yamlSpec string, seed int64, durationSec int) []Event {
	// minimal Phase 2 compiler: ignore yamlSpec details; emit one limit order per ms.
	r := rand.New(rand.NewSource(seed))
	out := make([]Event, 0, durationSec*1000)
	for i := 0; i < durationSec*1000; i++ {
		out = append(out, Event{
			OffsetNs: uint64(i) * 1_000_000,
			BotID:    uint64(r.Intn(8) + 1),
			Order: OrderSpec{
				Side:      []string{"bid","ask"}[r.Intn(2)],
				OrderType: "limit",
				TIF:       "gtc",
				Qty:       uint64(1 + r.Intn(20)),
				Price:     90 + int64(r.Intn(20)),
			},
		})
	}
	return out
}

func ScheduleHash(events []Event) string {
	h := sha256.New()
	for _, e := range events {
		var b [8]byte; binary.LittleEndian.PutUint64(b[:], e.OffsetNs); h.Write(b[:])
		binary.LittleEndian.PutUint64(b[:], e.BotID); h.Write(b[:])
		binary.LittleEndian.PutUint64(b[:], e.Qty); h.Write(b[:])
		binary.LittleEndian.PutUint64(b[:], uint64(e.Price)); h.Write(b[:])
		h.Write([]byte(e.Side)); h.Write([]byte(e.OrderType)); h.Write([]byte(e.TIF))
	}
	_ = fnv.New64()
	return string(h.Sum(nil))
}
```

- [ ] **Step 2: Dispatcher pushes events to bot-workers via simple HTTP**

`coordinator/dispatcher.go`:
```go
package coordinator

import (
	"bytes"
	"context"
	"encoding/json"
	"net/http"
	"time"
)

type Dispatcher struct { GatewayURL string; HTTP *http.Client }

func (d *Dispatcher) Run(ctx context.Context, events []Event) error {
	if d.HTTP == nil { d.HTTP = &http.Client{Timeout: 5 * time.Second} }
	start := time.Now()
	for _, e := range events {
		dt := time.Until(start.Add(time.Duration(e.OffsetNs)))
		if dt > 0 { time.Sleep(dt) }
		body, _ := json.Marshal(map[string]any{
			"bot_id": e.BotID, "local_seq": e.OffsetNs,
			"side": e.Order.Side, "qty": e.Order.Qty, "price": e.Order.Price,
			"order_type": e.Order.OrderType, "tif": e.Order.TIF,
		})
		req, _ := http.NewRequestWithContext(ctx, "POST", d.GatewayURL+"/v1/order", bytes.NewReader(body))
		req.Header.Set("Content-Type", "application/json")
		_, _ = d.HTTP.Do(req)  // Phase 2: ignore errors; Phase 3 measures them.
	}
	return nil
}
```

```rust
async fn main() -> std::io::Result<()> {
```
